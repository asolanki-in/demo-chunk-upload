import path from "path"; import fs from "fs-extra"; import { extractEntitlements, createDebugEntitlements } from "./entitlements"; import { resignApp } from "./codesign"; import { runCommand } from "./utils"; export async function resignAppOrIpa( inputPath: string, // Can be .ipa or .app cert: string, provisioningProfilePath: string, buildType: "debug" | "developer" | "distribution" ) { let appPath: string; const isIpa = inputPath.endsWith(".ipa"); const tempDir = path.join(__dirname, "../temp"); if (isIpa) { console.log("üì¶ Detected an IPA file, extracting..."); if (fs.existsSync(tempDir)) await fs.remove(tempDir); await fs.mkdir(tempDir, { recursive: true }); // Unzip IPA await runCommand(`unzip "${inputPath}" -d "${tempDir}"`); // Find .app inside the extracted `Payload/` const payloadPath = path.join(tempDir, "Payload"); const apps = await fs.readdir(payloadPath); const appFolder = apps.find((file) => file.endsWith(".app")); if (!appFolder) throw new Error("‚ùå No .app found inside IPA"); appPath = path.join(payloadPath, appFolder); } else { console.log("üìÇ Detected an .app bundle, using directly..."); appPath = inputPath; } // Handle entitlements const entitlementsPath = path.join(tempDir, "entitlements.plist"); let hasEntitlements = await extractEntitlements(appPath, entitlementsPath); if (!hasEntitlements && buildType === "debug") { await createDebugEntitlements(entitlementsPath); hasEntitlements = true; } await resignApp( appPath, cert, hasEntitlements ? entitlementsPath : null, provisioningProfilePath ); if (isIpa) { // Repackage IPA const resignedIpaPath = path.join(__dirname, `resigned-${buildType}.ipa`); await runCommand(`cd ${tempDir} && zip -r "${resignedIpaPath}" Payload`); await fs.remove(tempDir); console.log(`‚úÖ Resigned IPA: ${resignedIpaPath}`); } else { console.log(`‚úÖ Resigned .app successfully: ${appPath}`); } }
